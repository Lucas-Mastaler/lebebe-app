(async () => {
  console.clear()

  const OSRM_OFICIAL = 'https://osrm.lebebe.cloud'

  const cenarios = [
    {
      id: 'K13',
      nome: 'Cornelius — comparador legado x v2',
      payload: {
        cep: '81830-020',
        dataInicial: '2026-08-14',
        tempoNecessario: '00:40',
        destLat: -25.5091859,
        destLng: -49.2671477,
        destDisplay: 'Rua Cornelius Pries, 669, Xaxim, Curitiba - PR, 81830-020',
        destProvider: 'manual',
        enderecoCompleto: 'Rua Cornelius Pries, 669, Xaxim, Curitiba - PR, 81830-020'
      }
    },
    {
      id: 'K14',
      nome: 'Sítio Cercado — comparador legado x v2',
      payload: {
        cep: '81925-370',
        dataInicial: '2026-06-25',
        tempoNecessario: '00:40',
        destLat: -25.545418,
        destLng: -49.261836,
        destDisplay: 'Rua Attílio Silva Fonseca, 149-1 - Sítio Cercado, Curitiba - PR, 81925-370',
        destProvider: 'manual',
        enderecoCompleto: 'Rua Attílio Silva Fonseca, 149-1 - Sítio Cercado, Curitiba - PR, 81925-370'
      }
    },
    {
      id: 'K15',
      nome: 'Mandirituba — comparador legado x v2',
      payload: {
        cep: '83800-000',
        dataInicial: '2026-07-10',
        tempoNecessario: '00:40',
        destLat: -25.769705,
        destLng: -49.325586,
        destDisplay: 'R. José Schueda Sobrinho, 63-47 - Conj. Barcelona, Mandirituba - PR, 83800-000',
        destProvider: 'manual',
        enderecoCompleto: 'R. José Schueda Sobrinho, 63-47 - Conj. Barcelona, Mandirituba - PR, 83800-000'
      }
    }
  ]

  const km = m => typeof m === 'number' ? Number((m / 1000).toFixed(3)) : null

  const resumirCandidato = c => ({
    dataISO: c?.dataISO ?? c?.dataTexto,
    equipe: c?.equipe,
    tipo: c?.tipo,
    rank: c?.rank,
    horaMarcada: c?.horaMarcada,
    kmAdicionalNaRotaM: c?.kmAdicionalNaRotaM,
    kmAdicionalNaRotaKm: km(c?.kmAdicionalNaRotaM),
    origemKmAdicional: c?.origemKmAdicional
  })

  const contarTipos = candidatos => {
    const lista = candidatos ?? []

    return {
      total: lista.length,
      normais: lista.filter(c => c.tipo === 'normal').length,
      especiais: lista.filter(c => c.tipo === 'especial').length,
      premiums: lista.filter(c => c.tipo === 'premium').length,
      horaMarcada: lista.filter(c => c.horaMarcada === true).length
    }
  }

  const validarRegraExtrasUteis = candidatos => {
    const lista = (candidatos ?? [])
      .map(resumirCandidato)
      .filter(c => c.dataISO)

    const normais = lista.filter(c => c.tipo === 'normal')
    const extras = lista.filter(c => c.tipo !== 'normal')

    const ultimaNormalDataISO = normais
      .map(c => c.dataISO)
      .sort()
      .at(-1) ?? null

    const extrasInvalidos = ultimaNormalDataISO
      ? extras.filter(c => c.dataISO >= ultimaNormalDataISO)
      : []

    return {
      ultimaNormalDataISO,
      extrasTotal: extras.length,
      extrasInvalidosTotal: extrasInvalidos.length,
      extrasInvalidos,
      regraOk: extrasInvalidos.length === 0
    }
  }

  const resultados = []

  for (const cenario of cenarios) {
    console.log(`Rodando ${cenario.id} — ${cenario.nome}...`)

    const r = await fetch('/api/procurar-datas/v2/comparar', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        ...cenario.payload,
        osrmBaseUrlDiagnostico: OSRM_OFICIAL
      })
    })

    let body
    try {
      body = await r.json()
    } catch (err) {
      body = { erroParseJson: String(err) }
    }

    const legadoCandidatos = body?.legado?.resultadoNormalizado?.candidatos ?? []
    const v2Candidatos = body?.v2?.resultadoNormalizado?.candidatos ?? []
    const divergencias = body?.comparacao?.divergencias ?? []

    const legadoResumo = contarTipos(legadoCandidatos)
    const v2Resumo = contarTipos(v2Candidatos)
    const regraExtrasUteis = validarRegraExtrasUteis(v2Candidatos)

    const resultado = {
      id: cenario.id,
      nome: cenario.nome,

      status: r.status,
      httpOk: r.ok,
      ok: body?.ok,
      modo: body?.modo,

      legadoOk: body?.legado?.ok,
      v2Ok: body?.v2?.ok,

      tempoLegadoMs: body?.legado?.tempoMs,
      tempoV2Ms: body?.v2?.tempoMs,

      osrmBaseUrlUsadoV2:
        body?.diagnosticoMinimo?.osrmBaseUrlUsadoV2 ??
        body?.v2?.diagnosticoMinimo?.osrmBaseUrlUsado,

      osrmFallbackUsadoV2:
        body?.diagnosticoMinimo?.osrmFallbackUsadoV2 ??
        body?.v2?.diagnosticoMinimo?.osrmFallbackUsado,

      legadoResumo,
      v2Resumo,

      regraExtrasUteis,

      divergenciasResumo: {
        total: divergencias.length,
        criticas: divergencias.filter(d => d.severidade === 'critico').length,
        avisos: divergencias.filter(d => d.severidade === 'aviso').length,
        infos: divergencias.filter(d => d.severidade === 'info').length,
        tiposCriticos: divergencias.filter(d => d.severidade === 'critico').map(d => d.tipo),
        tiposAvisos: divergencias.filter(d => d.severidade === 'aviso').map(d => d.tipo),
        tiposInfos: divergencias.filter(d => d.severidade === 'info').map(d => d.tipo)
      },

      okOperacional:
        r.ok &&
        body?.ok === true &&
        body?.legado?.ok === true &&
        body?.v2?.ok === true &&
        !divergencias.some(d => d.severidade === 'critico') &&
        regraExtrasUteis.regraOk === true,

      legadoCandidatos: legadoCandidatos.map(resumirCandidato),
      v2Candidatos: v2Candidatos.map(resumirCandidato),
      divergencias
    }

    resultados.push(resultado)

    console.log(`${cenario.id} — RESUMO:`)
    console.log(JSON.stringify({
      id: resultado.id,
      nome: resultado.nome,
      status: resultado.status,
      ok: resultado.ok,
      legadoOk: resultado.legadoOk,
      v2Ok: resultado.v2Ok,
      tempoLegadoMs: resultado.tempoLegadoMs,
      tempoV2Ms: resultado.tempoV2Ms,
      legadoResumo: resultado.legadoResumo,
      v2Resumo: resultado.v2Resumo,
      regraExtrasUteis: resultado.regraExtrasUteis,
      divergenciasResumo: resultado.divergenciasResumo,
      okOperacional: resultado.okOperacional
    }, null, 2))

    console.log(`${cenario.id} — LEGADO:`)
    console.table(resultado.legadoCandidatos)

    console.log(`${cenario.id} — V2:`)
    console.table(resultado.v2Candidatos)

    console.log(`${cenario.id} — DIVERGÊNCIAS:`)
    console.table(resultado.divergencias.map(d => ({
      tipo: d.tipo,
      severidade: d.severidade,
      descricao: d.descricao
    })))
  }

  const resumoGeral = {
    okOperacional: resultados.every(r => r.okOperacional),
    totalCenarios: resultados.length,
    cenariosOkOperacional: resultados.filter(r => r.okOperacional).map(r => r.id),
    cenariosComCritico: resultados
      .filter(r => r.divergenciasResumo.criticas > 0)
      .map(r => r.id),
    cenariosComAviso: resultados
      .filter(r => r.divergenciasResumo.avisos > 0)
      .map(r => r.id),
    cenariosComExtraInvalido: resultados
      .filter(r => !r.regraExtrasUteis.regraOk)
      .map(r => r.id),
    compactos: resultados.map(r => ({
      id: r.id,
      status: r.status,
      ok: r.ok,
      legadoOk: r.legadoOk,
      v2Ok: r.v2Ok,
      quantidadeLegado: r.legadoResumo.total,
      quantidadeV2: r.v2Resumo.total,
      normaisV2: r.v2Resumo.normais,
      especiaisV2: r.v2Resumo.especiais,
      premiumsV2: r.v2Resumo.premiums,
      ultimaNormalDataISO: r.regraExtrasUteis.ultimaNormalDataISO,
      extrasInvalidosTotal: r.regraExtrasUteis.extrasInvalidosTotal,
      criticas: r.divergenciasResumo.criticas,
      avisos: r.divergenciasResumo.avisos,
      tempoLegadoMs: r.tempoLegadoMs,
      tempoV2Ms: r.tempoV2Ms,
      osrmFallbackUsadoV2: r.osrmFallbackUsadoV2,
      okOperacional: r.okOperacional
    }))
  }

  console.log('K18.5 — COMPARADOR LEGADO X V2 COM EXTRAS ÚTEIS — RESUMO GERAL:')
  console.log(JSON.stringify(resumoGeral, null, 2))

  console.log('K18.5 — COMPACTOS:')
  console.table(resumoGeral.compactos)

  window.__K18_5_COMPARADOR_EXTRAS_UTEIS__ = {
    resumoGeral,
    resultados
  }
})()
